B. Divisiblity of Differences

Hint ?

Think about the remainder!

Solution ?

For every remainder after under modulo m, check if any of them have more then $k$ elements in it.

Since the only way of constructing a set, which satisfies the condition that the difference between any two of them is divisible by m, is taking elements that has the same remainder.

 1#include <bits/stdc++.h>
 2
 3using namespace std;
 4
 5vector<int> rem[100100];
 6int main()
 7{
 8    int n, k, m;
 9    scanf("%d %d %d", &n, &k, &m);
10
11    for(int i = 0; i < n; i++) {
12        int num;
13        scanf("%d", &num);
14
15        rem[num % m].push_back(num);
16    }
17
18    // i = 0
19    for(int i = 0; i < m; i++) {
20        if((int)rem[i].size() >= k) {
21            printf("Yes\n");
22            for(int j = 0; j < k; j++)
23                printf("%d ", rem[i][j]);
24            return 0;
25        }
26    }
27    printf("No\n");
28
29    return 0;
30}

C. Classroom Watch

Hint ?

What are the possible values of the sum of digits of a given number?

Solution ?

The range of the sum of digits of a given number is $1$ ~ $9*9$ ($1$ ~ $999999999$).

Assume $x = n - sum\ of\ digits\ of\ n$, where $sum\ of\ digits\ of\ n$ is in range $[1, 81]$. Using the above property, we can check for all x + sum_of_digits(x) and see if any of them equals to n. If yes, add that the the answer set!

 1#include <bits/stdc++.h>
 2
 3using namespace std;
 4
 5int main()
 6{
 7    int n;
 8    scanf("%d", &n);
 9
10    set<int> ans;
11    for(int i = 1; i <= 81; i++) {
12        if(n - i >= 1) {
13            int num = n - i;
14            int sum = n - i;
15            while(num > 0) {
16                sum += num % 10;
17                num /= 10;
18            }
19
20            if(sum == n)
21                ans.insert(n - i);
22        }
23    }
24
25    printf("%d\n", (int)ans.size());
26    for(auto i : ans)
27        printf("%d ", i);
28
29    return 0;
30}

D. Sorting the Coins

Hint ?

What will the end result look like?

Solution ?

Since the end result will look like OOO...XXX..., and during the simulation one can observe that the total number of X in the O..X..O..X.. sequence, excluding the Xs in the end, is what will affect the hardness value. You can circle all XX..OOO and observe them. For every pass, total number of circled groups will decrease by 1.

For example, if we have OOOXOOXOXX, we will need 2 passes to adjust and 1 pass of checking. So the hardness value is $4 - 2 + 1$. 4 is the total number of X in the sequence, $-2$ is the Xs that is not in the end, and $+1$ is for the checking pass.

 1#include <bits/stdc++.h>
 2
 3using namespace std;
 4
 5struct UFDS {
 6    int par[300100];
 7    void init() {
 8        memset(par, -1, sizeof(par));
 9    }
10
11    int root(int x) {
12        return par[x] < 0 ? x : par[x] = root(par[x]);
13    }
14
15    void merge(int x, int y) {
16        x = root(x);
17        y = root(y);
18
19        if(x == y)
20            return;
21        if(par[x] > par[y])
22            swap(x, y);
23        par[x] += par[y];
24        par[y] = x;
25    }
26
27    int sz(int x) {
28        return -par[root(x)];
29    }
30} uf;
31
32int main()
33{
34    int n;
35    scanf("%d", &n);
36    uf.init();
37
38    bool has[n];
39    memset(has, false, sizeof(has));
40    printf("1 ");
41    for(int i = 1; i <= n; i++) {
42        int pos;
43        scanf("%d", &pos);
44        pos--;
45
46        if(pos - 1 >= 0) {
47            if(has[pos - 1] == true) {
48                uf.merge(pos - 1, pos);
49            }
50        }
51
52        if(pos + 1 < n) {
53            if(has[pos + 1] == true) {
54                uf.merge(pos + 1, pos);
55            }
56        }
57
58        has[pos] = true;
59        
60        if(has[n - 1] == false)
61            printf("%d ", i + 1);
62        else
63            printf("%d ", i - uf.sz(n - 1) + 1);
64    }
65
66    return 0;
67}
