挑題： 叢壯滋

題目的解法都不唯一，我只是照著我的習慣打了 sample code。

英哩轉公里 和 判斷座標是否在圓形的範圍內，是基本功力題，如果做對前兩題代表你會寫程式，也會基本IO。

遞迴程式練習，是遞迴基本題，如果不會寫的話請一定要花時間畫圖弄懂 recursive call graph。

各位數和排序 是排序題，也算是看看大家進階一點的C/C++語言使用能力。解法多元，大家可以多多討論學習。

過半元素 和 一整數序列所含之整數個數及平均值，都是IO題，加上一點基本演算法。如果可以秒殺，恭喜！

題目2. 英哩轉公里

Solution

%.1f 會自動四捨五入喔！

AC Code

1#include <bits/stdc++.h>
2
3int main()
4{
5	int n;
6	while(scanf("%d", &n) == 1)
7		printf("%.1f\n", n * 1.6);
8	return 0;
9}

題目3. 判斷座標是否在圓形的範圍內

Solution

$x^2 + y^2 = r^2$

這個公式大家記得嗎？

AC Code

 1#include <bits/stdc++.h>
 2
 3int main()
 4{
 5	int x, y;
 6	while(scanf("%d %d", &x, &y) == 2) {
 7		if(x * x + y * y <= 200 * 200)
 8			printf("inside\n");
 9		else
10			printf("outside\n");
11	}
12	return 0;
13}

題目12. 遞迴程式練習

Solution

遞迴的終止條件最重要了！

AC Code

 1#include <bits/stdc++.h>
 2
 3int f(int n)
 4{
 5    if (n == 0 || n == 1)
 6        return n + 1;
 7    return f(n - 1) + f(n / 2);
 8}
 9
10int main()
11{
12    int n;
13    while (scanf("%d", &n) == 1)
14        printf("%d\n", f(n));
15
16    return 0;
17}

題目26. 各位數和排序

Solution

這題做法很多！

可以開struct紀錄，也可以直接利用pair的特性。

最簡單的做法，直接在compare function做！

C++的compare function所利用的性質是如果$a$要在$b$前面就回傳true。

AC Code (using pair)

 1#include <bits/stdc++.h>
 2
 3using namespace std;
 4
 5typedef pair<int, int> ii;
 6
 7int getSum(int n)
 8{
 9    int sum = 0;
10    while (n > 0) {
11        sum += n % 10;
12        n /= 10;
13    }
14    return sum;
15}
16
17int main()
18{
19    int n;
20    scanf("%d", &n);
21
22    ii inp[n];
23    for (int i = 0; i < n; i++) {
24        int num;
25        scanf("%d", &num);
26
27        inp[i] = ii(getSum(num), num);
28    }
29
30    sort(inp, inp + n);
31    for (int i = 0; i < n; i++)
32        printf("%d%c", inp[i].second, i == n - 1 ? '\n' : ' ');
33
34    return 0;
35}

AC Code (using compare function)

 1#include <bits/stdc++.h>
 2
 3using namespace std;
 4
 5typedef pair<int, int> ii;
 6
 7int getSum(int n)
 8{
 9    int sum = 0;
10    while (n > 0) {
11        sum += n % 10;
12        n /= 10;
13    }
14    return sum;
15}
16
17bool cmp(const int &a, const int &b)
18{
19    int sa = getSum(a);
20    int sb = getSum(b);
21    if (sa == sb)
22        return a < b;
23    return sa < sb;
24}
25
26int main()

31    int inp[n];

34

37        printf("%d%c", inp[i], i == n - 1 ? '\n' : ' ');
38
39    return 0;
40}

題目31. 過半元素

Solution

讀取一行一筆測資，請多多研究字串！

fgets vs scanf。

解法不唯一，聽說過半元素有$O(n)$解法。

AC Code (strtok)

 1#include <bits/stdc++.h>
 2using namespace std;
 3
 4int main()
 5{
 6    char str[10000];
 7    while (fgets(str, 10000, stdin) != NULL) {
 8        char *num = strtok(str, "\n\r ");
 9        map<int, int> cnt;
10        int total = 0;
11        while (num != NULL) {
12            cnt[atoi(num)]++;
13            total++;
14            num = strtok(NULL, "\n\r ");
15        }
16
17        int ok = INT_MIN;
18        // 5 -> 3
19        // 6 -> 4
20        for (auto i : cnt)
21            if (i.second >= (total + 2) / 2) {
22                ok = i.first;
23            }
24
25        if (ok == INT_MIN)
26            printf("NO\n");
27        else
28            printf("%d\n", ok);
29    }
30    return 0;
31}

AC Code (getline + istringstream)

謝謝黃鈺程幫忙！

 1#include <bits/stdc++.h>
 2using namespace std;
 3#define sz(x) (int(x.size()))
 4int main()
 5{
 6    string inp;
 7    while (getline(cin, inp)) {
 8        istringstream iss(inp);
 9        vector<int> v;
10        int item;
11        while (iss >> item) {
12            v.push_back(item);
13        }
14        sort(v.begin(), v.end());
15        int res = v[sz(v) / 2];
16        int cnt = count(v.begin(), v.end(), res);
17        if (cnt > sz(v) / 2)
18            cout << res << endl;
19        else
20            cout << "NO" << endl;
21    }
22    return 0;
23}

題目33. 一整數序列所含之整數個數及平均值

Solution

字串讚讚！

AC Code

 1#include <bits/stdc++.h>
 2
 3int main()
 4{
 5    char inp[10000];
 6    while (fgets(inp, 10000, stdin) != NULL) {
 7        char *num = strtok(inp, "\r\n ");
 8        int total = 0;
 9        int sz = 0;
10        while (num != NULL) {
11            sz++;
12            total += atoi(num);
13
14            num = strtok(NULL, "\r\n ");
15        }
16
17        printf("Size: %d\n", sz);
18        printf("Average: %.3f\n", total * 1.0 / sz);
19    }
20    return 0;
21}
