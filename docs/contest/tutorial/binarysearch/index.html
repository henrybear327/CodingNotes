Vanya and Lanterns

Solution

有數學解，但是可以練習一下浮點數二分搜尋。

AC Code

 1#include <bits/stdc++.h>
 2
 3using namespace std;
 4
 5#define EPS 1e-10
 6
 7int n, len;
 8bool check(vector<int> &inp, double mid)
 9{
10    if (inp[0] - mid > -EPS)
11        return false;
12    double farthest = inp[0] + mid;
13    for (int i = 1; i < (int)inp.size(); i++) {
14        if (farthest - (inp[i] - mid) >= -EPS)
15            farthest = inp[i] + mid;
16        else
17            return false;
18    }
19
20    if (farthest - len >= -EPS)
21        return true;
22    return false;
23}
24
25int main()
26{
27    scanf("%d %d", &n, &len);
28
29    vector<int> inp;
30    for (int i = 0; i < n; i++) {
31        int num;
32        scanf("%d", &num);
33
34        inp.push_back(num);
35    }
36
37    sort(inp.begin(), inp.end());
38    auto it = unique(inp.begin(), inp.end());
39    inp.resize(distance(inp.begin(), it));
40
41    double l = 0, r = len;
42    for (int i = 0; i < 200; i++) {
43        double mid = (l + r) / 2;

挑戰題

Grapevine

進階題，大家可以挑戰一下。

AC Code

 1#include <bits/stdc++.h>
 2
 3// For every query, binary search for the upper-left corner of the square
 4// And then binary search for the lower-left corner "on the diagnal
 5
 6using namespace std;
 7
 8int n, m;
 9int inp[555][555];
10int low, high;
11
12int search(int row, int col)
13{
14    // printf("row %d col %d\n", row, col);
15    int l = -1, r = min(n - row, m - col);
16    while(r - l > 1) {
17        int mid = l + (r - l) / 2;
18        // printf("l = %d, mid = %d, r = %d\n", l, mid, r);
19        
20        if(inp[row + mid][col + mid] <= high)
21            l = mid;
22        else
23            r = mid;
24    }
25    // printf("l = %d\n", l);
26    
27    return l;
28}
29
30void solve()
31{
32    int ret = 0;
33    for(int i = 0; i < n; i++) {
34        int l = lower_bound(inp[i], inp[i] + m, low) - inp[i];
35        if(l == m)
36            continue;
37        int r = search(i, l);
38        if(r == -1)
39            continue;
40        // printf("i = %d l = %d r = %d\n", i, l, r);
41        
42        ret = max(ret, r + 1);
43    }
44    
45    printf("%d\n", ret);
46}
47
48int main()
49{
50    while(scanf("%d %d", &n, &m) == 2 && (n || m)) {
51        for(int i = 0; i < n; i++)
52            for(int j = 0; j < m; j++) 
53                scanf("%d", &inp[i][j]);
54        
55        int q;
56        scanf("%d", &q);
57        for(int i = 0; i < q; i++) {
58            scanf("%d %d", &low, &high);
59            
60            solve();
61        }
62        printf("-\n");
63    }
64    
65    return 0;
66}

  5	public static boolean check(int[] inp, long mid, int k) {
  6		int[] cur = Arrays.copyOf(inp, inp.length);
  7
  8		int last = inp.length - 1;
  9		while (last >= 0 && cur[last] == 0)
 10			last--;
 11
 12		for (int i = 0; i < k; i++) {
 13			long t = mid - last - 1;
 14			while (t > 0 && last >= 0) {
 15				if (t - cur[last] >= 0) { // move box
 16					t -= cur[last];
 17					cur[last] = 0;
 18
 19					while (last >= 0 && cur[last] == 0)
 20						last--;
 21				} else {

 24				}
