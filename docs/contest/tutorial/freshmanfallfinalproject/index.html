     1#include <stdio.h>
 2
 3int check(int ans, int guess)
 4{
 5    int ansDecompose[4] = {0}, guessDecompose[4] = {0};
 6    int aCnt[10] = {0}, gCnt[10] = {0};
 7    int orig = guess;
 8	
 9	int i;
10    for (i = 0; i < 4; i++) {
11        ansDecompose[i] = ans % 10;
12        aCnt[ans % 10]++;
13        ans /= 10;
14    }
15    for (i = 0; i < 4; i++) {
16        guessDecompose[i] = guess % 10;
17        if (gCnt[guess % 10] != 0) {
18            printf("%d is an invalid guess.\n", orig);
19            return 0;
20        }
21        gCnt[guess % 10]++;
22        guess /= 10;
23    }
24
25    int a = 0, b = 0;
26
27    for (i = 0; i < 4; i++) {
28        if (ansDecompose[i] == guessDecompose[i])
29            a++;
30        else if (aCnt[guessDecompose[i]] > 0) {
31            b++;
32        }
33    }
34
35    printf("%dA%dB\n", a, b);
36
37    return a == 4;
38}

40void solve()
41{
42    int ans;
43    scanf("%d", &ans);
44
45    int guess;
46
47    int history[9999] = {0};
48    while (scanf("%d", &guess) == 1) {
49        // check guessing history
50        if (history[guess] == 1) {
51            printf("%d is already guessed.\n", guess);
52            continue;
53        }
54        history[guess] = 1;
55
56        if (check(ans, guess) == 1)
57            break;
58    }
59}

64    scanf("%d", &ncase);

66        solve();
67        if (ncase)
68            printf("\n");

71    return 0;

    要如何把四個方向有效率的實作移動呢？沒錯，就是透過旋轉！

    只要實作其中一個方向的移動後，其他方向就是先選轉到剛剛實作好的方向後，做完，再轉回去就好啦！

    注意 Game over 條件，別看錯了QQ

 13int checkSame(int inp[4][4])
 14{
 15    int i, j, k;
 16    int hasSame = 0;
 17    for (i = 0; i < 4; i++) {
 18        for (j = 0; j < 4; j++) {
 19            if (inp[i][j] == 0)
 20                continue;
 21            for (k = j + 1; k < 4; k++) {
 22                if (inp[i][k] == 0)
 23                    continue;
 24                if (inp[i][j] == inp[i][k]) {
 25                    hasSame = 1;
 26                } else {
 27                    j = k - 1;
 28                    break;
 29                }
 30            }
 31        }
 32    }
 33    return hasSame;
 34}

 36void rotate90Left(int inp[4][4])
 37{
 38    int i, j, tmp[4][4];
 39    for (i = 0; i < 4; i++)
 40        for (j = 0; j < 4; j++) {
 41            tmp[i][j] = inp[j][4 - 1 - i];
 42        }
 43
 44    for (i = 0; i < 4; i++)
 45        for (j = 0; j < 4; j++)
 46            inp[i][j] = tmp[i][j];
 47}

 49// 0, no message
 50// 1, has 64
 51// 2, no move
 52int move(int inp[4][4])
 53{
 54    // printf("Before\n");
 55    // print(inp);
 56
 57    int i, j, k;
 58    int has0 = 0;
 59    int has64 = 0;
 60    for (i = 0; i < 4; i++) {
 61        for (j = 0; j < 4; j++) {
 62            if (inp[i][j] == 0)
 63                continue;
 64
 65            for (k = j + 1; k < 4; k++) {
 66                if (inp[i][k] == 0) // 0, continue
 67                    continue;
 68                if (inp[i][j] == inp[i][k]) { // same, merge
 69                    inp[i][j] += inp[i][k];
 70                    inp[i][k] = 0;
 71                    break;
 72                } else { // not the same, skip
 73                    break;
 74                }
 75            }
 76        }
 77    }
 78
 79    for (i = 0; i < 4; i++) {
 80        int lastZero = -1;
 81        for (j = 0; j < 4; j++) {
 82            if (inp[i][j] == 64)
 83                has64 = 1;
 84            if (inp[i][j] == 0)
 85                has0 = 1;
 86
 87            if (inp[i][j] != 0 && lastZero != -1) {
 88                inp[i][lastZero] = inp[i][j]; // move
 89                inp[i][j] = 0;                // fill zero
 90                lastZero = j;                 // crucial
 91                continue;
 92            }
 93            if (inp[i][j] == 0 &&
 94                lastZero == -1) // update last when no previous zero is present
 95                lastZero = j;
 96        }
 97    }
 98
 99    // printf("after\n");
100    // print(inp);
101
102    if (has64 == 1)
103        return 1;
104    else if (has64 == 0 && has0 == 0) {
105        int hasSame = checkSame(inp);
106        if (hasSame == 0) {
107            rotate90Left(inp);
108            hasSame = checkSame(inp);
109            rotate90Left(inp);
110            rotate90Left(inp);
111            rotate90Left(inp);
112        }
113        return hasSame == 0 ? 2 : 0;
114    } else
115        return 0;
116}

125void solve()
126{
127    int i, j, inp[4][4];
128    for (i = 0; i < 4; i++)
129        for (j = 0; j < 4; j++)
130            scanf("%d", &inp[i][j]);
131    char dir[10];
132    scanf("%s", dir);
133
134    // a left
135    // s down
136    // d right
137    // w up
138
139    // merge and move
140    int ret = 0;
141    if (dir[0] == 'a') {
142        ret = move(inp);
143    } else if (dir[0] == 'd') {
144        rotate90Left(inp);
145        rotate90Left(inp);
146        ret = move(inp);
147        rotate90Left(inp);
148        rotate90Left(inp);
149    } else if (dir[0] == 's') {
150        rotate90Left(inp);
151        rotate90Left(inp);
152        rotate90Left(inp);
153        ret = move(inp);
154        rotate90Left(inp);
155    } else { // w
156        rotate90Left(inp);
157        ret = move(inp);
158        rotate90Left(inp);
159        rotate90Left(inp);
160        rotate90Left(inp);
161    }
162
163    print(inp);
164    if (ret == 1)
165        printf("You win\n");
166    else if (ret == 2)
167        printf("Game over\n");
168}
